# Lineplot of the average revenue by the different sales methods, by week
with sns.axes_style("whitegrid"):
    weekly_sales_lineplot = sns.relplot(data=avg_revenue_weekly_sales, kind='line', x='week', y='mean', 
                                       hue='sales_method', marker='P', clip_on=False)
    
    # Title
    weekly_sales_lineplot.fig.suptitle("Figure 4. Average Revenue Generated by Sales Methods per Week", y=1.04)
    
    # Tick markers
    weekly_sales_lineplot.set(ylim=(0, 250), xlim=(1, 6))
    
    # Axes labels
    weekly_sales_lineplot.set_axis_labels("Week", "Average Revenue")
    
    # Legend
    sns.move_legend(weekly_sales_lineplot, "center right", bbox_to_anchor=(1.04, 0.5))
